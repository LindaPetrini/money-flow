  <behavior>
    detectBankFormat:
    - headers including 'TransferWise ID' → 'wise'
    - headers including 'Amount (EUR)' → 'n26'
    - headers including 'Started Date' → 'revolut'
    - headers with none of the above → 'unknown'
    - BOM prefix on first header ('\ufeffTransferWise ID') still → 'wise' (PapaParse strips BOM with header:true, but guard anyway)

    parseEuropeanAmount:
    - '1.234,56' → 1234.56
    - '-847,50' → -847.5
    - '-12.50' → -12.5  (dot decimal — unchanged; removing dots when comma present is safe)
    - '0,00' → 0
    - '-1.234.567,89' → -1234567.89 (multiple dot thousand-seps)

    extractExpenses (Wise):
    - rows: [{Date:'2024-01-15', Amount:'-42.50', Currency:'EUR', Description:'Grocery'}, ...]
    - excludes rows where Currency !== 'EUR'
    - excludes rows where Amount >= 0 (income/transfers)
    - returns ParsedTransaction with date='2024-01-15', description='Grocery', amountEur=-42.5

    extractExpenses (N26):
    - column: 'Amount (EUR)'; negative = expense
    - date column: 'Date'
    - description from 'Payee' column (fallback to 'Payment reference')
    - excludes rows where Amount (EUR) >= 0

    extractExpenses (Revolut):
    - column: 'Amount'; negative = expense
    - date column: 'Started Date' — format 'YYYY-MM-DD HH:MM:SS', extract YYYY-MM-DD prefix
    - description from 'Description' column
    - excludes rows where Amount >= 0 OR State !== 'COMPLETED'

    extractExpenses (unknown):
    - returns []

    parseCSVFile:
    - calls Papa.parse with { header: true, skipEmptyLines: true }
    - BOM stripped automatically by PapaParse with header:true
    - detects format from parsed header keys
    - calls extractExpenses
    - returns sorted by date descending (newest first)

    ```typescript
    function parseEuropeanAmount(raw: string): number {
      const trimmed = raw.trim();
      // If comma present: dots are thousand separators, comma is decimal
      if (trimmed.includes(',')) {
        return parseFloat(trimmed.replace(/\./g, '').replace(',', '.'));
      }
      // No comma: standard dot decimal (Wise, Revolut)
      return parseFloat(trimmed);
    }
    ```
