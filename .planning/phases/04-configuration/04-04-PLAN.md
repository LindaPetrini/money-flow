**File 1: src/features/settings/OverflowRatiosSection.tsx**

Named export `OverflowRatiosSection`.

**State design:**
- `localRatios: Array<{ accountId: string; pctStr: string }>` — mirrors the persisted ratios but with pct as a string for editing.
- Initialize from `settings.overflowRatios` when the component mounts and when that value changes (use a separate useEffect with `settings.overflowRatios` as dependency, but only update localRatios if not dirty — see below).
- `dirty: boolean` — true once user has made edits; prevents the useEffect from overwriting mid-edit.

**Total calculation:**
```typescript
const total = localRatios.reduce((sum, r) => {
  const n = parseFloat(r.pctStr);
  return sum + (isNaN(n) ? 0 : n);
}, 0);
const isValid = Math.round(total) === 100;
```

**Save logic:**
```typescript
const handleSave = async () => {
  const ratios: OverflowRatio[] = localRatios.map(r => ({
    accountId: r.accountId,
    pct: parseFloat(r.pctStr) || 0,
  }));
  await updateSettings({ overflowRatios: ratios });
  setDirty(false);
};
```

**Add row:** A dropdown "Add account" select shows accounts NOT yet in localRatios. Selecting an account appends `{ accountId: selected, pctStr: '0' }` to localRatios.

**Delete row:** Remove row from localRatios.

**Layout:**
```
[Account name]   [pct input]%   [Delete button]
...
Total: {total}%  ← shows in red if != 100, green if == 100

[Save button — disabled when !isValid]  [Revert button — resets to persisted ratios]
```

If no accounts are configured, show: "No accounts configured — add accounts in the Accounts tab first."

**Full skeleton:**

import { useState, useEffect } from 'react';
import { useSettingsStore } from '@/stores/settingsStore';
import { useAccountStore } from '@/stores/accountStore';
import type { OverflowRatio } from '@/types/domain';

export function OverflowRatiosSection() {
  const settings = useSettingsStore(s => s.settings);
  const updateSettings = useSettingsStore(s => s.updateSettings);
  const accounts = useAccountStore(s => s.accounts);

  const [localRatios, setLocalRatios] = useState<Array<{ accountId: string; pctStr: string }>>(
    settings.overflowRatios.map(r => ({ accountId: r.accountId, pctStr: String(r.pct) }))
  );
  const [dirty, setDirty] = useState(false);

  // Sync from store when not dirty
  useEffect(() => {
    if (!dirty) {
      setLocalRatios(settings.overflowRatios.map(r => ({
        accountId: r.accountId,
        pctStr: String(r.pct),
      })));
    }
  }, [settings.overflowRatios, dirty]);

  // ... total, isValid, handleSave, handleRevert, handleAdd, handleDelete, handlePctChange
  // handleRevert: setLocalRatios(persisted snapshot), setDirty(false)

  const accountsNotInRatios = accounts.filter(
    a => !localRatios.some(r => r.accountId === a.id)
  );

**Full skeleton:**
```tsx
import { useSettingsStore } from '@/stores/settingsStore';
import { useAccountStore } from '@/stores/accountStore';
import { parseCents } from '@/lib/cents';
import type { Cents } from '@/lib/cents';
import { useState } from 'react';

export function TaxBufferSection() {
  const settings = useSettingsStore(s => s.settings);
  const updateSettings = useSettingsStore(s => s.updateSettings);
  const accounts = useAccountStore(s => s.accounts);

  const [taxPctStr, setTaxPctStr] = useState(String(settings.taxPct));
  const [taxAccountId, setTaxAccountId] = useState(settings.taxAccountId);
  const [bufferTargetStr, setBufferTargetStr] = useState(
    (settings.bufferTargetCents / 100).toFixed(2)
  );
  const [bufferAccountId, setBufferAccountId] = useState(settings.bufferAccountId);

  const taxPctNum = parseFloat(taxPctStr);
  const taxPctValid = !isNaN(taxPctNum) && taxPctNum >= 0 && taxPctNum <= 100;

  const handleSave = async () => {
    if (!taxPctValid) return;
    await updateSettings({
      taxPct: taxPctNum,
      taxAccountId,
      bufferTargetCents: parseCents(bufferTargetStr) as number,
      bufferAccountId,
    });
  };

  return (
    <div className="space-y-8">
      {/* Tax section */}
      <div className="space-y-4">
        <h2 className="text-sm font-semibold">Tax Settings</h2>
        {/* tax % input */}
        {/* tax account select */}
      </div>

      {/* Buffer section */}
      <div className="space-y-4">
        <h2 className="text-sm font-semibold">Buffer Settings</h2>
        {/* buffer target input */}
        {/* buffer account select */}
      </div>

      <button onClick={handleSave} disabled={!taxPctValid} ...>
        Save
      </button>
    </div>
  );
}
```

**Input styling:** `text-sm border border-border rounded px-2 py-1 bg-background w-24` for number inputs; `text-sm border border-border rounded px-2 py-1 bg-background` for selects.

**Field label style:** `text-sm text-muted-foreground` in a `flex items-center gap-3` row with the input.
